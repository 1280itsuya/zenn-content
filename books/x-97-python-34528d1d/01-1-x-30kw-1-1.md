## 通知ログを取得するPlaywright収集スクリプト（Python 3.12対応）

X API v2無料枠は通知エンドポイントへのアクセスが不可なため、Playwrightでブラウザを操作してHTMLを取得した。

```python
# collect_notifications.py
import asyncio, json
from datetime import datetime
from playwright.async_api import async_playwright

COOKIES_PATH = "x_cookies.json"  # 第2章で取得手順を詳述

async def collect(limit: int = 500) -> list[dict]:
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        ctx = await browser.new_context()
        with open(COOKIES_PATH) as f:
            await ctx.add_cookies(json.load(f))

        page = await ctx.new_page()
        await page.goto("https://x.com/notifications/mentions")

        items, seen = [], set()
        while len(items) < limit:
            tweets = await page.query_selector_all('[data-testid="tweet"]')
            for t in tweets:
                text = await t.inner_text()
                h = hash(text[:80])
                if h not in seen:
                    seen.add(h)
                    items.append({"text": text, "ts": datetime.now().isoformat()})

            await page.evaluate("window.scrollBy(0, 2000)")
            await asyncio.sleep(1.5)
            if len(items) >= limit:
                break

        await browser.close()
        return items

if __name__ == "__main__":
    data = asyncio.run(collect())
    with open("notifications_raw.json", "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    print(f"取得: {len(data)}件")
```

事前に `pip install playwright && playwright install chromium` が必要。

```python
import json, re

NOISE_PATTERNS = [
    r"副\s*業", r"不労所得", r"在宅", r"稼[げぐ]", r"月\d+万",
```

## X公式フィルタが届かない3パターン：完全一致・文脈無視・属性非考慮

```python
# x公式が弾けない実例
examples = [
    "AIで副　業するなら今がチャンス",        # ① 全角スペース挿入
    "この本読んで不労所♡得 目指す！",        # ② 絵文字挿入
    "月収１０万円突破した話（スレッド）",      # ③ 全角数字
    "在宅で仕事してる人に聞きたいんだが",     # ④ 文脈違い（誤検知方向）
    "詳しくはプロフィール欄から",             # ⑤ 部分一致が微妙にズレる
]

for e in examples:
    hit = any(kw in e for kw in ["副業", "不労所得", "在宅ワーク", "詳細はプロフ"])
    print(f"{'BLOCK' if hit else 'PASS '} | {e}")
```


## 自分の通知汚染度を5分で数値化する診断スクリプト

```python
# noise_audit.py  ← この章だけで動く診断ツール
import json, re, sys
from pathlib import Path

def audit(path: str):
    data = json.loads(Path(path).read_text(encoding="utf-8"))
    texts = [d["text"] if isinstance(d, dict) else d for d in data]

    SIGNS = [
        r"稼[げぐ]", r"副\s*業", r"月\d+万", r"在宅",
        r"プロフ[ィイ]", r"LINE", r"無料", r"今だけ", r"限定", r"不労"
    ]
    flagged = sum(1 for t in texts if any(re.search(p, t) for p in SIGNS))
    rate = flagged / len(texts) * 100 if texts else 0

    print(f"総件数       : {len(texts)}")
    print(f"ノイズ推定   : {flagged} ({rate:.1f}%)")
    label = "深刻(>80%)" if rate > 80 else "要対策(>50%)" if rate > 50 else "許容範囲"
    print(f"汚染スコア   : {label}")

if __name__ == "__main__":
    audit(sys.argv[1] if len(sys.argv) > 1 else "notifications_raw.json")
```

手順：`collect_notifications.py` で JSON を取得 → `python noise_audit.py notifications_raw.json` を実行 → 汚染スコアを確認。
