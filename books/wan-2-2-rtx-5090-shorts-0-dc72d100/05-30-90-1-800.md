# YouTube Analytics API で実測値を引っ張るスクリプト
from googleapiclient.discovery import build
from google.oauth2.credentials import Credentials

def fetch_shorts_metrics(channel_id: str) -> list:
    creds = Credentials.from_authorized_user_file("token.json")
    yt = build("youtubeAnalytics", "v2", credentials=creds)
    response = yt.reports().query(
        ids=f"channel=={channel_id}",
        startDate="2026-04-01",
        endDate="2026-04-30",
        metrics="views,estimatedMinutesWatched,averageViewPercentage,subscribersGained",
        dimensions="video",
        sort="-views",
        maxResults=90
    ).execute()
    return response["rows"]

rows = fetch_shorts_metrics("UCxxxxxxxxxx")
top7 = [r for r in rows if r[1] >= 10000]
avg_completion = sum(r[3] for r in rows) / len(rows)
print(f"インプレ1万超え: {len(top7)}本 / 平均完了率: {avg_completion:.1f}%")

**Kling I2V APIで実写風素材を追加する拡張構成**

Wan 2.2の弱点は「実在する人物・動物の自然な動き」。Kling I2V APIに実写写真を渡すことで、Wan生成と実写映像を混在させたShortsが作れる。

```python
import httpx, base64, time, os

def generate_kling_i2v(image_path: str, prompt: str) -> str:
    with open(image_path, "rb") as f:
        img_b64 = base64.b64encode(f.read()).decode()

    headers = {
        "Authorization": f"Bearer {os.environ['KLING_API_KEY']}",
        "Content-Type": "application/json",
    }
    payload = {
        "model_name": "kling-v1-5",
        "image": img_b64,
        "prompt": prompt,
        "duration": "5",
        "aspect_ratio": "9:16",
    }
    r = httpx.post(
        "https://api.klingai.com/v1/videos/image2video",
        json=payload, headers=headers, timeout=30
    )
    task_id = r.json()["data"]["task_id"]

    for _ in range(60):
        time.sleep(10)
        status = httpx.get(
            f"https://api.klingai.com/v1/videos/image2video/{task_id}",
            headers=headers
        ).json()
        if status["data"]["task_status"] == "succeed":
            return status["data"]["task_result"]["videos"][0]["url"]
    raise TimeoutError("Kling I2V: 10分以内に完了しなかった")

video_url = generate_kling_i2v("cat_photo.jpg", "cat walking slowly on tatami floor")
print(f"生成完了: {video_url}")
```

コスト試算：Kling I2V 5秒クリップ ≈ $0.14（編集部試算・2026年6月レート）、月90本中30本に適用で約¥600。Wan 2.2ローカル生成（0円）と組み合わせ、実写比率を30%以下に抑えることでコストを最小化しながらCTRを底上げできる。
